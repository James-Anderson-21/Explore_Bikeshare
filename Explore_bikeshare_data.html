# This function takes in a data frame and city name and returns the 5 most popular starts and their count
common_starts = function(df, City.Name) {
    city_data = df[df$City == City.Name, ]
    station_counts = table(city_data$Start.Station)
    station_counts = data.frame(sort(station_counts, decreasing = TRUE)[1:5])
    names(station_counts) = c(paste('Start.Station'), ('Count'))
    return(station_counts)
}
